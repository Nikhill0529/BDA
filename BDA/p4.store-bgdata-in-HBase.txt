#Step 2: Import Libraries
import happybase
import pandas as pd
______________________________________________________________

#Step 3: Connect to HBase Server
connection = happybase.Connection('localhost')
print(connection.tables())
______________________________________________________________

#Step 4: Create HBase Table
connection.create_table(
 'student',
 {'info': dict()}
)
________________________________________________________________


#Step 5: Insert Data into Table
table = connection.table('student')
table.put('101', {
 'info:name': 'Amit',
 'info:course': 'Data Science',
 'info:marks': '85'
})
table.put('102', {
 'info:name': 'Neha',
 'info:course': 'AI',
 'info:marks': '90'
})
table.put('103', {
 'info:name': 'Rahul',
 'info:course': 'Big Data',
 'info:marks': '78'
})
______________________________________________________________________

#Step 6: Retrieve Data from HBase
for key, data in table.scan():
 print(key, data)
_______________________________________________________________________

#Step 7: Update Data
table.put('101', {'info:marks': '88'})

________________________________________________________________________

#Step 8: Delete Data
table.delete('103')
