#Step 4: Upload Dataset to HDFS (Simulation)
!mkdir input
!cp student_data.csv input/
!ls input

_________________________________________________________________________________________
#Step 5: Create Pig Script
%%writefile student.pig
student_data = LOAD 'input/student_data.csv'
USING PigStorage(',')
AS (id:int, name:chararray, course:chararray, marks:int);
-- Display dataset
DUMP student_data;
-- Filter students with marks greater than 85
high_marks = FILTER student_data BY marks > 85;
DUMP high_marks;

__________________________________________________________________________________________
#Step 6: Run Pig Script
!pig pig-0.17.0/bin/pig student.pig
